law_corporatist_labour_syndicate = {
	group = lawgroup_labor_rights
	icon = "gfx/interface/icons/law_icons/law_corporatist_syndicate.png"
	
	progressiveness = 0

	unlocking_technologies = { corporatism }
	
	modifier = { 
		state_expected_sol_from_literacy = 1
	}
	
	institution = institution_workplace_safety
	institution_modifier = {
		state_radicals_from_sol_change_mult = -0.05
		state_loyalists_from_sol_change_mult = 0.1
		interest_group_ig_trade_unions_pop_attraction_mult = -0.06
		building_minimum_wage_mult = -0.06
		state_expected_sol_from_literacy = -1
	}
	
	on_activate = {
		recalculate_pop_ig_support = yes
	}
	on_deactivate = {
		recalculate_pop_ig_support = yes
	}
	
	possible_political_movements = {
		law_regulatory_bodies
		law_worker_protections
		law_union_representation
	}
	
	pop_support = {
		value = 0
		add = {
			desc = "POP_CAPITALISTS"			
			if = {
				limit = {
					owner = {
						OR = { 
							has_law = law_regulatory_bodies
							has_law = law_type:law_union_representation
							has_law = law_type:law_worker_protections
						}
					}
					is_pop_type = capitalists
				}
				value = 1
			}
		}
	}

	can_impose = {
		is_subject_of = scope:initiator	
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}

law_regulatory_bodies = {
	group = lawgroup_labor_rights
	
	icon = "gfx/interface/icons/law_icons/regulatory_bodies.dds"
	
	progressiveness = 50

	unlocking_technologies = {
		labor_movement
	}

	institution = institution_workplace_safety

	institution_modifier = {
		building_minimum_wage_mult = 0.075
		tax_dividends_add = 0.025
		state_poor_standard_of_living_add = 0.25
	}
	
	possible_political_movements = {
		law_worker_protections
	}
	
	pop_support = {
		value = 0

		add = {
			desc = "POP_DANGEROUS_JOBS"		
			if = {
				limit = { 
					NOT = { strata = rich }
					OR = {
						pop_employment_building_group = bg_mining
						pop_employment_building_group = bg_manufacturing
					}
					owner = { 
						NOT = { has_law = law_type:law_worker_protections }
					}
				}
				add = {
					value = literacy_rate
					divide = 2
				}
			}			
		}
	}

	on_impose = { 
			
	} 

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}
