# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law
# cultural_acceptance_rule and religious_acceptance_rule are triggers called in culture scope with scope:country as the country with the laws

law_ethnostate = {
	group = lawgroup_citizenship
	
	icon = "gfx/interface/icons/law_icons/ethnostate.dds"
	
	progressiveness = -75
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}
	
	cultural_acceptance_rule = {
		is_primary_culture_of = scope:country
	}

	unlocking_technologies = {
		nationalism
	}
	
	modifier = {
		country_authority_add = 300
		state_radicals_from_sol_change_accepted_culture_mult = -0.3
		state_loyalists_from_sol_change_accepted_culture_mult = 0.3
		state_accepted_birth_rate_mult = 0.15
		interest_group_ig_petty_bourgeoisie_pop_attraction_mult = 0.4
		state_radicals_from_discrimination_mult = 0.5
		country_infamy_generation_mult = 0.1
		country_bureaucracy_investment_cost_factor_mult = -0.35
	}
	
	possible_political_movements = {
		law_racial_segregation
		law_cultural_exclusion
	}

	on_activate = {
		ig:ig_petty_bourgeoisie ?= {
			remove_ideology = ideology_reactionary_exclusionist
			remove_ideology = ideology_reactionary_segregationist
			remove_ideology = ideology_reactionary_multicultural
		}
	}

	pop_support = {
		value = 0
	}

	on_impose = { 
			
	} 

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}
