communist_rebelist = {
	color = { 255 0 0 }
	name = {
		first_valid = {
			triggered_desc = {
				desc = mass_association_spartacusbund
				trigger = {
						OR = {
							country_has_primary_culture = cu:north_german
							country_has_primary_culture = cu:south_german
						}
					}
				}
			triggered_desc = {
				desc = mass_association_jacobin
				trigger = {
					exists = c:FRA
					THIS = c:FRA
				}
			}				
			triggered_desc = {
				desc = mass_association_bolshevik
				trigger = {
					country_has_primary_culture = cu:russian
					}
				}				
			triggered_desc = {
				desc = mass_association_wrrh
				trigger = {
						OR = {
							country_has_primary_culture = cu:han
							country_has_primary_culture = cu:hakka
							country_has_primary_culture = cu:miao
							country_has_primary_culture = cu:min
							country_has_primary_culture = cu:zhuang
							country_has_primary_culture = cu:yi
							country_has_primary_culture = cu:yue
					}
				}
			}
			triggered_desc = {
				desc = mass_association_rebelist
				trigger = {
					always = yes
				}
			}			
		}
	}
	icon = {
		default = "/gfx/interface/icons/political_parties/communists.dds"
	}
	valid_for_country = {
		always = yes
	}
	available_for_interest_group = {
		owner = {
		has_law = law_type:law_council_republic
			OR = {
			has_law = law_type:law_mass_democracy 
			has_law = law_type:law_anarchy 
			}	
		any_interest_group = {
				leader = {
				    OR = {
						has_ideology = ideology:ideology_vanguardist
						has_ideology = ideology:ideology_left_communism		
						has_ideology = ideology:ideology_communist						
					}	
				}
			}
		}	
		NOR = {
			is_interest_group_type = ig_industrialists
			is_interest_group_type = ig_landowners
		}
	     	NOT = {
	         	owner = {
	         		has_variable = only_valid_party
	         	}
	        }		
	}
	join_weight = {
		value = {
			value = -5
			desc = "base_value"
		}
		if = { # vanguardist love rebelist
			limit = {
				leader = {
				OR = {
						has_ideology = ideology:ideology_vanguardist
						has_ideology = ideology:ideology_communist						
					}	
				}
			}
			add = {
				value = 10
				desc = "from_communism"
			}
		}
		
		if = { # left communism like rebelist
			limit = {
				leader = {
						has_ideology = ideology:ideology_left_communism						
				}
			}
			add = {
				value = 6
				desc = "from_communism"
			}
		}
		
		if = { # left communists love rebelist when have democratic planed economy
			limit = {
				owner = {
					has_law = law_type:law_communal_principle
				}			
				leader = {
						has_ideology = ideology:ideology_left_communism						
				}
			}
			add = {
				value = 9
				desc = "from_left_communism"
			}
		}
		
		if = { # far_lefts hate rebelist when don't have democratic economy
			limit = {
				owner = {
					NOR = {
					has_law = law_type:law_communal_principle
					has_law = law_type:law_cooperative_ownership			
					}
				}			
				leader = {
					OR = {
					    has_ideology = ideology:ideology_anarchist
					    has_ideology = ideology:ideology_syndicalism
						has_ideology = ideology:ideology_left_communism	
					}	
				}
			}
			add = {
				value = -11
				desc = "from_left_communism_hate"
			}
		}
		
		if = { # left communists love continuous revolution
			limit = {
				owner = {
					ruler = {
					    OR = {
						has_ideology = ideology:ideology_vanguardist				
						has_ideology = ideology:ideology_left_communism						
						}
					}
				}			
				leader = {
						has_ideology = ideology:ideology_left_communism						
				}
			}
			add = {
				value = 6
				desc = "from_continuous_revolution"
			}
		}
		
		if = { # far_left hate bureaucracy
			limit = {
				owner = {
					NOT = {has_law = law_type:law_communal_adminstration}
				}			
				leader = {
				OR = {
						has_ideology = ideology:ideology_left_communism
					    has_ideology = ideology:ideology_anarchist
					    has_ideology = ideology:ideology_syndicalism						
                    }						
				}
			}
			add = {
				value = -7
				desc = "from_hate_bureaucracy"
			}
		}

		if = { # moderates hate far left
			limit = {
				owner = {
					ruler = {
					    OR = {
						has_ideology = ideology:ideology_left_communism
					    has_ideology = ideology:ideology_anarchist
					    has_ideology = ideology:ideology_syndicalism					
						}
					}
				}			
				leader = {
				OR = {
						has_ideology = ideology:ideology_communist
                        has_ideology = ideology:ideology_social_democrat
						has_ideology = ideology:ideology_technocratic_socialism							
					}	
				}
			}
			add = {
				value = -6
				desc = "from_left_radical"
			}
		}
		
		if = { # syndicalist conservatism
			limit = {
				owner = {
					has_law = law_type:law_communal_adminstration
				}			
				leader = {
						has_ideology = ideology:ideology_syndicalism					
				}
			}
			add = {
				value = 14
				desc = "from_communal_adminstration"
			}
		}		
	}
}
