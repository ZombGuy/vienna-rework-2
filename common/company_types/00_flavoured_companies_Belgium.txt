	}
	
	potential = {
		has_interest_marker_in_region = region_rhine
	}
	
	attainable = {

				OR = {
					state_region = s:STATE_WALLONIA

				}
			}
		}
	}	
	
	possible = {
		any_scope_state = {

	}

	}
}
