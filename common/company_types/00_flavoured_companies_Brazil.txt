	}
	
	potential = {
		has_dlc_feature = rp1_content
		has_interest_marker_in_region = region_brazil
	}
	
	attainable = {

		any_scope_state = {

		building_chemical_plants_throughput_add = 0.05
	}
	
	ai_weight = {

	}
}
