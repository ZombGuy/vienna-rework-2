je_kra_survey = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = survey_kra_taken
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		remove_modifier = modifier_surveying_kra
		trigger_event = { id = canal_events.101 }
	}

	current_value = {
		value = root.var:survey_kra_taken
	}

	goal_add_value = {
		value = 30
	}

	invalid = { # If the Kra Canal is being built and you're not the one building it, you've blown it, chief.
		OR = {
			NOT = { has_variable = survey_kra_taken }
			any_state = {
				state_region = s:STATE_TENASSERIM
				NOT = { owner = ROOT }
				any_scope_building = {
					is_building_type = building_kra_canal
				}
			}
		}
	}

	on_invalid = {
		remove_variable = survey_kra_taken
		remove_modifier = modifier_surveying_kra
	}

	progressbar = yes

	weight = 10
	should_be_pinned_by_default = yes
}

je_kra_canal = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	complete = {
		any_scope_state = {
			state_region = s:STATE_TENASSERIM
			any_scope_building = {
				is_building_type = building_kra_canal
				occupancy >= 0.9
			}
		}
	}

	on_complete = {
		set_global_variable = kra_canal_complete
	}
	should_be_pinned_by_default = yes
}
